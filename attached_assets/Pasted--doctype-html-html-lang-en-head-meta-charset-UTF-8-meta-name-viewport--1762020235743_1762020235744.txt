    <script>
      document.getElementById("chat-form").addEventListener("submit", function (e) {
        e.preventDefault();
        const input = document.getElementById("user-input");
        const chatHistory = document.querySelector(".chat-history");
        const userMessage = input.value.trim();
        if (userMessage) {
          const userP = document.createElement("p");
          userP.style.textAlign = "right";
          userP.style.color = "#fff";
          userP.textContent = "You: " + userMessage;
          chatHistory.appendChild(userP);
          setTimeout(() => {
            const aiP = document.createElement("p");
            aiP.style.textAlign = "left";
            aiP.style.color = "var(--accent-primary)";
            aiP.textContent = 'Concierge AI: I received your query! Please click "Book a Call" for detailed assistance.';
            chatHistory.appendChild(aiP);
            chatHistory.scrollTop = chatHistory.scrollHeight;
          }, 800);
          input.value = "";
          chatHistory.scrollTop = chatHistory.scrollHeight;
        }
      });
